### Kalkulator Koreksi Natrium (Double Method)

Gunakan kalkulator ini untuk mendapatkan parameter lengkap natrium pada pasien hiperglikemia.

<script>
const naIn = document.getElementById('na_input');
const gluIn = document.getElementById('glu_input');

function updateResults() {
  const na = parseFloat(naIn.value);
  const glu = parseFloat(gluIn.value);
  
  if (isNaN(na) || isNaN(glu)) return;

  const deltaGlu = Math.max(0, (glu - 100) / 100);
  
  const katz = na + (1.6 * deltaGlu);
  const hillier = na + (2.4 * deltaGlu);

  document.getElementById('res_katz').innerText = katz.toFixed(1);
  document.getElementById('res_hillier').innerText = hillier.toFixed(1);
}

function copyAll() {
  const na = naIn.value;
  const glu = gluIn.value;
  const katz = document.getElementById('res_katz').innerText;
  const hillier = document.getElementById('res_hillier').innerText;
  const btn = document.getElementById('btn-copy-all');

  const summaryText = `[Koreksi Natrium]\n` +
                      `- Na Terukur: ${na} mEq/L\n` +
                      `- GDS: ${glu} mg/dL\n` +
                      `---\n` +
                      `- Hasil Katz (1.6): ${katz} mEq/L\n` +
                      `- Hasil Hillier (2.4): ${hillier} mEq/L`;

  navigator.clipboard.writeText(summaryText).then(() => {
    const originalText = btn.innerText;
    btn.innerText = "Copied to Clipboard!";
    btn.style.background = "#4caf50";
    setTimeout(() => {
      btn.innerText = originalText;
      btn.style.background = "#333";
    }, 2000);
  });
}

naIn.addEventListener('input', updateResults);
gluIn.addEventListener('input', updateResults);
window.onload = updateResults;
</script>

### Cara Membaca Hasil
Teks yang disalin akan memiliki format sebagai berikut:
> **[Koreksi Natrium]**
> - Na Terukur: 130 mEq/L
> - GDS: 500 mg/dL
> ---
> - Hasil Katz (1.6): 136.4 mEq/L
> - Hasil Hillier (2.4): 139.6 mEq/L
